set(TESTS
    MicroX86DisTest
)

set(TESTS_WITH_ELF
    elf/ElfBinaryLoaderTest
)

    pe/Win32BinaryLoaderTest
)

        NAME ${t}
        SOURCES ${t}.h ${t}.cpp

    )
endforeach()

            SOURCES ${t}.h ${t}.cpp
            LIBRARIES
                ${DEBUG_LIB}
                boomerang
                ${CMAKE_THREAD_LIBS_INIT}
        )
    endforeach()
endif (BOOMERANG_BUILD_LOADER_Win32)
